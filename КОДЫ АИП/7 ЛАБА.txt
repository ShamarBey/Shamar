#include<locale.h>
#include<Windows.h>
using namespace std;

int main() 
{
    setlocale(LC_ALL, "Russian");
    SetConsoleCP(1251);
    SetConsoleOutputCP(1251);
    int N;
    double A, B;

    cout << "Введите номер действия (1-сложение, 2-вычитание, 3-умножение, 4-деление): ";
    cin >> N;

    cout << "Введите число A: ";
    cin >> A;

    cout << "Введите число B: ";
    cin >> B;

    if (N < 1 or N > 4) 
    {
        cout << "Неверный номер действия." << endl;
        return 1;
    }

    if (N == 4 or B == 0) 
    {
        cout << "Деление на ноль." << endl;
        return 1;
    }

    double result;
    switch (N) 
    {
    case 1:
        result = A + B;
        break;
    case 2:
        result = A - B;
        break;
    case 3:
        result = A * B;
        break;
    case 4:
        result = A / B;
        break;
    }

    cout << "Результат: " << result << endl;

    return 0;
}
